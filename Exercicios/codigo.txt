int max_diff(int a[], int n){
    int dif=0;
    for (int i=0; i<n-1; i++){
        if (a[i]<0 && a[i+1]<0){
            if (-a[i]+a[i+1]>dif){
                dif=-a[i]+a[i+1];
            }
            if (-a[i+1]+a[i]>dif){
                dif=-a[i+1]+a[i];
            }
        }
        if (a[i]>0 && a[i+1]<0){
            if (a[i]-a[i+1]>dif){
                dif=a[i]-a[i+1];
            }
        }
        if (a[i]<0 && a[i+1]>0){
            if (-a[i]+a[i+1]>dif){
                dif=-a[i]+a[i+1];
            }
        }
        if (a[i]>0 && a[i+1]>0){
            if (a[i]-a[i+1]>dif){
                dif=a[i]-a[i+1];
            }
            if (a[i+1]-a[i]>dif){
                dif=a[i+1]-a[i];
            }
        }
    }
    return dif;
}

        if (str[i] >= 'a' && str[i]<='z'){
            int valor=str[i]+k;

        return 1;
    } else if (snake[linha][coluna]=='#'){
        return 0;
    }

    snake[linha][coluna]='#';

    if (racheal(snake, r, c, linha+1, coluna)==1){
        return 1;
    }
    if (racheal(snake, r, c, linha, coluna+1)==1){
        return 1;
    }
    if (racheal(snake, r, c, linha-1, coluna)==1){
        return 1;
    }
    if (racheal(snake, r, c, linha, coluna-1)==1){
        return 1;
    }

    return 0;
}

void playlist(int musicas[], int i, int n, int d, int sum, int *best) {
    if (i == n) {
        if (sum > *best)
            *best = sum;
        return;
    }

    playlist(musicas, i+1, n, d, sum, best);

    if (sum + musicas[i] <= d) {
        playlist(musicas, i+1, n, d, sum + musicas[i], best);
    }
}

int main(void) {
    int d, n;
    int best = 0;
    int musicas[20];

    scanf("%d %d", &d, &n);

    for (int i = 0; i < n; i++) {
        scanf("%d", &musicas[i]);
    }

    playlist(musicas, 0, n, d, 0, &best);

    printf("%d\n", best);
    return 0;
}

int unique_letters (char str[]){
    int n=strlen(str);
    int contador;
    for (int i=0; i<n; i++){
        char letra_atual=str[i];
        if (!isalpha(letra_atual)){
            continue;
        }
        if (!(islower(letra_atual))){
            letra_atual=tolower(letra_atual);
        }

        int repetida=0;
        for (int j=0; j<i; j++){
            char letra_anterior;
            if (!(islower(letra_anterior))){
                letra_anterior=tolower(letra_anterior);
            }
            if (letra_anterior==letra_atual){
                repetida++;
                break;
            }
        }
        if (repetida==0){
            contador++;
        }
    }
    return contador;
}



Aqui tens o teu **Kit de Sobrevivência de Emergência**, só com a sintaxe pura:

### 1. O Esqueleto (Começa sempre por aqui)
```c
#include <stdio.h>
#include <string.h> // Se houver texto
#include <ctype.h>  // Se precisares de isalpha(), tolower(), etc.

int main(void) {
    // Código entra aqui
    return 0;
}
```

### 2. Ler e Imprimir (A regra do `&`)
* **Números (`int`):** TEM `&` no `scanf`
  ```c
  scanf("%d", &n);
  printf("%d\n", n);
  ```
* **Palavras (`string`):** NÃO TEM `&` no `scanf`
  ```c
  scanf("%s", palavra);
  printf("%s\n", palavra);
  ```

### 3. O Ciclo Universal (Para Arrays/Vetores)
Se te derem uma lista de números `a` com tamanho `n`, escreve logo isto:
```c
for (int i = 0; i < n; i++) {
    // Fazes qualquer coisa com a[i]
}
```

### 4. O Ciclo das Strings (Para Textos)
Se não souberes o tamanho da palavra, procura pelo `\0` (o fim da palavra):
```c
for (int i = 0; str[i] != '\0'; i++) {
    // Fazes qualquer coisa com a letra str[i]
}
```

### 5. O Duplo Ciclo (Para Matrizes e Grelhas)
Sempre que o problema falar em "linhas", "colunas", "tabuleiro" ou "mapa":
```c
for (int linha = 0; linha < r; linha++) {
    for (int coluna = 0; coluna < c; coluna++) {
        // Fazes qualquer coisa com grelha[linha][coluna]
    }
}
```

### 6. Pílulas Salva-Vidas (Decora isto)
* **Descobrir o Maior / Menor num array:**
  Começa o maior com um número minúsculo, e o menor com um gigante.
  ```c
  int max = -999999;
  int min = 999999;
  ```
* **Comparar Palavras:**
  NÃO uses `==`. Usa `strcmp`. Se der `0`, são iguais.
  ```c
  if (strcmp(palavra1, palavra2) == 0) { ... }
  ```
* **É letra minúscula?**    
  ```c
  if (letra >= 'a' && letra <= 'z') { ... }
  ```
* **Recursividade (Função que se chama a si própria):**
  Escreve **sempre** o `if` de paragem antes de qualquer outra coisa!
  ```c
  if (linha < 0 || coluna < 0 || linha >= r || coluna >= c) {
      return 0; // Saiu da grelha, abortar!
  }
  ```
